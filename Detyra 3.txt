using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Detyra3
{ 
 internal class Program
    {
        static void Main(string[] args)
        {
            
            Console.WriteLine("Vendosni nje fjalekalim:");
            string password  = Console.ReadLine();
            bool isUpper = password.Any(char.IsUpper);
            bool isLower = password.Any(char.IsLower);
            bool hasInt = password.Any(char.IsDigit);
            bool symbolCheck = password.Any(p => !char.IsLetterOrDigit(p));
            if (isUpper && isLower && hasInt && symbolCheck)
            {
                Console.WriteLine("password i sakte ");
            }
            else if (!isUpper)
            {
                Console.WriteLine("ska upper");
            }
            else if (!isLower)
            {
                Console.WriteLine("ska lower");
            }
            else if (!hasInt)
            {
                Console.WriteLine("ska nr");
            }
            else
            {
                Console.WriteLine("ska karakter special");
            }
            Console.ReadLine();
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Ushtrim3_3
{
    class Program
    {
        static void Main(string[] args)
        {
            string msg, str1, str2;
            int index1, index2;
            Console.WriteLine("Jepni mesazhin :");
            msg = Console.ReadLine();
            Console.WriteLine("Jepni pjesen 1 :");
            str1 = Console.ReadLine();
            Console.WriteLine("Jepni pjesen e 2:");
            str2 = Console.ReadLine();
            index1 = msg.ToLower().IndexOf(str1.ToLower());
            index2 = msg.ToLower().IndexOf(str2.ToLower());

            if (index1 != -1 && index2 != -1)
            {
                if (index1 > index2)
                {
                    int temp = index2;
                    index2 = index1;
                    index1 = temp;
                    string temp2 = str1;
                    str1 = str2;
                    str2 = temp2;
                }

                msg = msg.Substring(0, index1) + msg.Substring(index2, str2.Length) + msg.Substring(index1 + str1.Length, index2 - (index1 + str1.Length)) +
                msg.Substring(index1, str1.Length) + msg.Substring(index2 + str2.Length);
                Console.WriteLine(msg);
                
            }
            else Console.WriteLine("Fjalet nuk ndodhen ne stringe.");

            Console.ReadLine();
        }
    }
}
